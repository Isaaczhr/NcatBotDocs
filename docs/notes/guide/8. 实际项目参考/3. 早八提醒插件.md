### 介绍

这是一个用于演示[定时任务](../6.%20开发%20NcatBot%20插件/4.%20插件高级功能/4.5%20定时任务.md)的插件.

### 源代码

```python
from ncatbot.plugin.loader import BasePlugin, CompatibleEnrollment

import datetime

class ZaoBa(BasePlugin):
    name = "ZaoBa" # 插件名称
    version = "0.0.1" # 插件版本

    async def on_load(self):
        # 插件加载时执行的操作, 可缺省

        print(f"早八问候插件已加载")
        print(f"插件版本: {self.version}")
        self.add_scheduled_task(self.zaoba, "早八问候", "08:00", max_runs=10, args=("早八人", ))
        self.add_scheduled_task(self.remind, "起床提醒", "30s", max_runs=10, args_provider=lambda: (datetime.datetime.now().hour, datetime.datetime.now().minute))
        self.add_scheduled_task(self.exam, "考试提醒", "2025-03-27 13:12:20", max_runs=1, kwargs={"subject":"物理"})
    
    def zaoba(self, extra):
        print("你好, 早八人")
        print(extra)
    
    def remind(self, hour, minute):
        print(f"起床了, 已经 {hour} 点 {minute} 分了")
        
    def exam(self, subject):
        print(f"要考试了, 是 {subject}")
```
